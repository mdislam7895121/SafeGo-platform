   const toggleTripType = (code: DriverTripTypeCode) => {
     setEnabledTypes(prev => {
       if (prev.includes(code)) {
         const next = prev.filter(c => c !== code);
         // Prevent disabling ALL trip types – at least one must remain enabled.
         if (next.length === 0) return prev;
         return next;
       }
       return [...prev, code];
     });
   };
